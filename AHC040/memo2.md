## 現状のコード(mainのみ)

```
int main(){
    LL(n,t,sgm);
    v(w,n,0);v(h,n,0);
    rep(n) cin >> w[i] >> h[i];
    ll areaSum = 0;
    rep(n){
        areaSum += w[i] * h[i];
    }
    ll length = sqrt(areaSum);
    ll cnt = 0;
    bool flag = true;
    rep(j,t){
        cout << n << endl;
        rep(i,n){
            cout << i << spa;
            cnt += max(w[i], h[i]);
            if(w[i] < h[i]) cout << 1 << spa;
            else cout << 0 << spa;
            cout << "U" << spa;
            if(flag){
                cout << "-1" << endl;
                flag = false;
            }else{
                cout << i-1 << endl;
            }
            if(cnt > length){
                flag = true;
                cnt = 0;
            }
        }
        cout << endl;
    }
}
```
